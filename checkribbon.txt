{
  const frames = Array.from(document.querySelectorAll('iframe'));
  let editorFrame = null;
  for (const f of frames) {
    try {
      if (f.contentWindow?.document?.getElementById('xmlEditor')) {
        editorFrame = f;
        break;
      }
    } catch (e) { /* cross-origin, skip */ }
  }
  
  if (!editorFrame) {
    console.error('No frame contains xmlEditor. Try the DevTools frame dropdown — switch to the editor frame and run the direct snippet I gave you.');
  } else {
    console.log('Editor is in frame:', editorFrame.id);
    const w = editorFrame.contentWindow;
    const xml = w.document.getElementById('xmlEditor').value;
    const doc = new DOMParser().parseFromString(xml, 'application/xml');
    const cas = Array.from(doc.getElementsByTagName('CustomAction'));
    
    console.log('CustomActions in diff:', cas.length);
    cas.forEach((ca, i) => {
      const cui = ca.getElementsByTagName('CommandUIDefinition')[0];
      const ctrl = cui ? Array.from(cui.childNodes).find(c => c.nodeType === 1) : null;
      const menuButtons = ctrl ? Array.from(ctrl.getElementsByTagName('Button')) : [];
      console.log(`[${i}] ${ca.getAttribute('Id')}`);
      console.log(`    Location=${ca.getAttribute('Location')}`);
      console.log(`    Sequence=${ca.getAttribute('Sequence')}`);
      console.log(`    Control=${ctrl?.tagName} id=${ctrl?.getAttribute('Id')}`);
      console.log(`    Children (${menuButtons.length}):`);
      menuButtons.forEach(b => console.log(`      - ${b.getAttribute('Id')}`));
    });
    
    console.log('\nCommandDefinitions:', Array.from(doc.getElementsByTagName('CommandDefinition')).map(c => c.getAttribute('Id')));
    console.log('\n--- full XML ---\n' + xml);
  }
}
